    CERINTA:

       Se citeste de la tastatura o matrice patratica cu n linii si coloane cu elemente numere naturale nenule. Sa se modifice valorile divizibile cu 2 in 0 si cele divizibile cu 5 in -1. Daca un numar este divizibil cu ambele sa se modifice cu -2.

    REZOLVARE:

     #include < iostream >
     using namespace std;
     int main (){
     int a[100][100],n,i,j;
     cin>> n;
     for(i=1;i<=n;i++)
          for(j=1;j<=n;j++)
               cin>>a[i][j];
          for(i=1;i<=n;i++)
               for(j=1;j<=n;j++)
               if((a[i][j]%2==0)&&(a[i][j]%5==0))
                    a[i][j]=-2;
               else if(a[i][j]%2==0)
                    a[i][j]=0;
               else if(a[i][j]%5==0)
                    a[i][j]=-1;
          for(i=1;i<=n;i++){
               for(j=1;j<=n;j++)
                    cout<< a[i][j]<< " ";
                    cout<< endl;
          }
     return 0;
     }
