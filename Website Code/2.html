    CERINTA:

       Se citeste un vector de la tastatura cu n elemente numere naturale nenule. Sa se stearga din vector toate numerele prime.

    REZOLVARE:

     #include < iostream >
     using namespace std;
     int main (){
     int v[100],d,i,nr,n,j;
     cin>>n;
     for(i=1;i<=n;i++)
          cin<< v[i];
     for(i=1;i<=n;i++){
          nr=0;
          if(v[i]==1)nr=1;
          for(d=2;d<=v[i]/2;d++){
               if(v[i]%d==0)
                    nr++;
          }
          if(nr==0){
               for(j=i;j<=n;j++)
                    v[j]=v[j+1];
               n--;
               i--;
               }
          }
          for(i=1;i<=n;i++)
               cout<< v[i]<< " ";
          return 0;
          }
