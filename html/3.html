    CERINTA:

       Se citesc numere natural de la tastatura. Cand se intalneste un numar prim,programul sa se opreasca si Sa se afiseze mesajul "Gata".

    REZOLVARE:

     #include < iostream >
     using namespace std;
     int main (){
     int n,d;
     bool ok=false;
     while(ok==false){
          cin>> n;
          ok=true;
          for(d=2;(d*d<=n)&&(ok==true);d++)
               if(n%d==0)
               ok=false;
     }
     cout<< "Gata";
     return 0;
     }
