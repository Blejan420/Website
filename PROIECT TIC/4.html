    REZOLVARE:

     #include < iostream >
     using namespace std;
     int main (){
     int a,d,p,k;
     cin>> a;
     for(d=2;d<=a/2;d++){
          if(a%d==0)
          for(p=2;p<=d/2;p++)
               if(d%p==0)k++
               if(k==0)cout<< d<< ” ”;
          }
     return 0;
     }
