    REZOLVARE:

     #include < iostream >
     using namespace std;
     int main (){
     int n, putere , d=2;
     cin>> n;
     while(n>1){
          putere=0;
          while(n%d==0){
               putere++;
               n/=10;}
          if(putere!=0)
          cout<< d<< ”la puterea ”<< putere;
          d++;
          }
     return 0;
     }
